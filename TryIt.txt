#!/bin/bash

# Function to check if Java is installed
check_java() {
    if type -p java > /dev/null; then
        return 0  # Java is installed
    else
        return 1  # Java is not installed
    fi
}

# Function to open HTML page in the default web browser
open_html_page() {
    # Replace "/path/to/your/html/page.html" with the actual path to your HTML page
    html_page_url="file:///path/to/your/html/page.html"
    xdg-open "$html_page_url"  # Open HTML page in default web browser
}

# Main script logic
if check_java; then
    echo "Java is installed. Opening HTML page..."
    open_html_page
else
    echo "Error: Java is not installed."
fi
